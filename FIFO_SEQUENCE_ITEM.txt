`include "uvm_macros.svh"
class fifo_seq_item extends uvm_sequence_item;
    // Parameters (aligned with fifo_if.sv)
    parameter DATA_WIDTH = 16;

    // Transaction fields
    rand bit reset_n;
    rand bit write_en;
    rand bit read_en;
    rand logic [15:0] data_in; // Matches DATA_WIDTH = 16
    logic [15:0] data_out;
    bit full;
    bit empty;
    bit almost_full;
    bit almost_empty;
    bit overflow;
    bit underflow;

    // Constraint to restrict simultaneous write and read (optional, can be overridden)
    constraint write_read_exclusive {
        soft !(write_en && read_en); // Soft constraint to allow override in simultaneous read/write tests
    }

    // UVM macros for field automation
    `uvm_object_utils_begin(fifo_seq_item)
        `uvm_field_int(reset_n, UVM_ALL_ON)
        `uvm_field_int(write_en, UVM_ALL_ON)
        `uvm_field_int(read_en, UVM_ALL_ON)
        `uvm_field_int(data_in, UVM_ALL_ON)
        `uvm_field_int(data_out, UVM_ALL_ON)
        `uvm_field_int(full, UVM_ALL_ON)
        `uvm_field_int(empty, UVM_ALL_ON)
        `uvm_field_int(almost_full, UVM_ALL_ON)
        `uvm_field_int(almost_empty, UVM_ALL_ON)
        `uvm_field_int(overflow, UVM_ALL_ON)
        `uvm_field_int(underflow, UVM_ALL_ON)
    `uvm_object_utils_end

    // Constructor
    function new(string name = "fifo_seq_item");
        super.new(name);
    endfunction
endclass
